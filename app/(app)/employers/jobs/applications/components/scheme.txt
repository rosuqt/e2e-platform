create table public.job_postings (
  employer_id uuid not null,
  job_title character varying(255) not null,
  location character varying(255) null,
  remote_options character varying(50) null,
  work_type character varying(50) null,
  pay_type character varying(50) null,
  pay_amount character varying(50) null,
  recommended_course character varying(255) null,
  job_description text null,
  job_summary text null,
  must_have_qualifications text[] null,
  nice_to_have_qualifications text[] null,
  application_deadline timestamp without time zone null,
  max_applicants integer null,
  perks_and_benefits text[] null,
  verification_tier character varying(50) null default 'basic'::character varying,
  created_at timestamp without time zone null default now(),
  responsibilities character varying null,
  id uuid not null default gen_random_uuid (),
  paused boolean null default false,
  constraint job_postings2_pkey primary key (id),
  constraint job_postings_id_key unique (id),
  constraint job_postings2_employer_id_fkey foreign KEY (employer_id) references registered_employers (id) on delete CASCADE
) TABLESPACE pg_default;  create table public.applications (
  application_id uuid not null default gen_random_uuid (),
  student_id uuid null,
  job_id uuid not null,
  experience_years character varying not null,
  portfolio character varying null,
  resume character varying not null,
  cover_letter character varying null,
  terms_accepted boolean not null,
  first_name character varying not null,
  last_name character varying not null,
  email character varying not null,
  phone character varying not null,
  address character varying not null,
  application_questions jsonb null,
  application_answers jsonb not null,
  describe_proj text null,
  status character varying null,
  constraint application_answers_pkey primary key (application_id),
  constraint application_answers_job_id_fkey foreign KEY (job_id) references job_postings (id) on delete CASCADE
) TABLESPACE pg_default;
